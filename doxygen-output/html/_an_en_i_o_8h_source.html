204  template<typename T>
205  errorType
206  read_vector_(std::string var_name, std::vector<T> & vector) const {
207 
208  using namespace netCDF;
209  using namespace std;
210 
211  NcFile nc(file_path_, NcFile::FileMode::read);
212  NcVar var = nc.getVar(var_name);
213  auto var_dims = var.getDims();
214  T *p_vals = nullptr;
215 
216  size_t len = 1;
217  for (const auto & dim : var_dims) {
218  len *= dim.getSize();
219  }
220 
221  try {
222  p_vals = new T[len];
223  } catch (bad_alloc & e) {
224  nc.close();
225  if (verbose_ >= 1) cout << BOLDRED <<
226  "Error: Insufficient memory when reading variable ("
227  << var_name << ")!" << RESET << endl;
228  return (INSUFFICIENT_MEMORY);
229  }
230 
231  var.getVar(p_vals);
232  vector.assign(p_vals, p_vals + len);
233 
234  nc.close();
235  delete [] p_vals;
236  return (SUCCESS);
237  };
