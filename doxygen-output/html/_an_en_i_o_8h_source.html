455  template<typename T>
456  errorType
457  read_vector_(std::string var_name, std::vector<T> & results) const {
458 
459  using namespace netCDF;
460  using namespace std;
461 
462  if (mode_ != "Read") {
463  if (verbose_ >= 1) cout << "Error: Mode should be 'Read'." << endl;
464  return (WRONG_MODE);
465  }
466 
467  NcFile nc(file_path_, NcFile::FileMode::read);
468 
469  NcVar var = nc.getVar(var_name);
470  if (var.isNull()) {
471  if (verbose_ >= 1) cout << BOLDRED << "Error: Could not"
472  << " find variable " << var_name << "!" << RESET << endl;
473  return (WRONG_INDEX_SHAPE);
474  }
475 
476  auto var_dims = var.getDims();
477  T *p_vals = nullptr;
478 
479  size_t len = 1;
480  for (const auto & dim : var_dims) {
481  len *= dim.getSize();
482  }
483 
484  try {
485  p_vals = new T[len];
486  } catch (bad_alloc & e) {
487  nc.close();
488  if (verbose_ >= 1) cout << BOLDRED
489  << "Error: Insufficient memory when reading variable ("
490  << var_name << ")!" << RESET << endl;
491  nc.close();
492  return (INSUFFICIENT_MEMORY);
493  }
494 
495  var.getVar(p_vals);
496  results.assign(p_vals, p_vals + len);
497 
498  nc.close();
499  delete [] p_vals;
500  return (SUCCESS);
501  };

503  template<typename T>
504  errorType
505  read_vector_(std::string var_name, std::vector<T> & results,
506  std::vector<size_t> start, std::vector<size_t> count,
507  std::vector<ptrdiff_t> stride = {1}) const {
508 
509  using namespace netCDF;
510  using namespace std;
511 
512  if (mode_ != "Read") {
513  if (verbose_ >= 1) cout << "Error: Mode should be 'Read'." << endl;
514  return (WRONG_MODE);
515  }
516 
517  NcFile nc(file_path_, NcFile::FileMode::read);
518  NcVar var = nc.getVar(var_name);
519  T *p_vals = nullptr;
520 
521  size_t total = 1;
522  for (auto i : count) {
523  total *= i;
524  }
525 
526  try {
527  p_vals = new T[total];
528  } catch (bad_alloc & e) {
529  nc.close();
530  if (verbose_ >= 1) cout << BOLDRED
531  << "Error: Insufficient memory when reading variable ("
532  << var_name << ")!" << RESET << endl;
533  nc.close();
534  return (INSUFFICIENT_MEMORY);
535  }
536 
537  // Reverse the order because of the storage style of NetCDF
538  reverse(start.begin(), start.end());
539  reverse(count.begin(), count.end());
540  reverse(stride.begin(), stride.end());
541 
542  var.getVar(start, count, stride, p_vals);
543  results.assign(p_vals, p_vals + total);
544 
545  nc.close();
546  delete [] p_vals;
547  return (SUCCESS);
548  };
