374  template<typename T>
375  errorType
376  read_vector_(std::string var_name, std::vector<T> & results) const {
377 
378  using namespace netCDF;
379  using namespace std;
380 
381  if (mode_ != "Read") {
382  if (verbose_ >= 1) cout << "Error: Mode should be 'Read'." << endl;
383  return (WRONG_MODE);
384  }
385 
386  NcFile nc(file_path_, NcFile::FileMode::read);
387  NcVar var = nc.getVar(var_name);
388  auto var_dims = var.getDims();
389  T *p_vals = nullptr;
390 
391  size_t len = 1;
392  for (const auto & dim : var_dims) {
393  len *= dim.getSize();
394  }
395 
396  try {
397  p_vals = new T[len];
398  } catch (bad_alloc & e) {
399  nc.close();
400  if (verbose_ >= 1) cout << BOLDRED
401  << "Error: Insufficient memory when reading variable ("
402  << var_name << ")!" << RESET << endl;
403  nc.close();
404  return (INSUFFICIENT_MEMORY);
405  }
406 
407  var.getVar(p_vals);
408  results.assign(p_vals, p_vals + len);
409 
410  nc.close();
411  delete [] p_vals;
412  return (SUCCESS);
413  };

417  read_vector_(std::string var_name, std::vector<T> & results,
418  std::vector<size_t> start, std::vector<size_t> count,
419  std::vector<ptrdiff_t> stride = {1}) const {
420 
421  using namespace netCDF;
422  using namespace std;
423 
424  if (mode_ != "Read") {
425  if (verbose_ >= 1) cout << "Error: Mode should be 'Read'." << endl;
426  return (WRONG_MODE);
427  }
428 
429  NcFile nc(file_path_, NcFile::FileMode::read);
430  NcVar var = nc.getVar(var_name);
431  T *p_vals = nullptr;
432 
433  size_t total = 1;
434  for (auto i : count) {
435  total *= i;
436  }
437 
438  try {
439  p_vals = new T[total];
440  } catch (bad_alloc & e) {
441  nc.close();
442  if (verbose_ >= 1) cout << BOLDRED
443  << "Error: Insufficient memory when reading variable ("
444  << var_name << ")!" << RESET << endl;
445  nc.close();
446  return (INSUFFICIENT_MEMORY);
447  }
448 
449  // Reverse the order because of the storage style of NetCDF
450  reverse(start.begin(), start.end());
451  reverse(count.begin(), count.end());
452  reverse(stride.begin(), stride.end());
453 
454  var.getVar(start, count, stride, p_vals);
455  results.assign(p_vals, p_vals + total);
456 
457  nc.close();
458  delete [] p_vals;
459  return (SUCCESS);
460  };
