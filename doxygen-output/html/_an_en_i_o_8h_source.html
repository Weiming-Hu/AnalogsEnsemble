315  template<typename T>
316  errorType
317  read_vector_(std::string var_name, std::vector<T> & vector) const {
318 
319  using namespace netCDF;
320  using namespace std;
321 
322  if (mode_ != "Read") {
323  if (verbose_ >= 1) cout << "Error: Mode should be 'Read'." << endl;
324  return (WRONG_MODE);
325  }
326 
327  NcFile nc(file_path_, NcFile::FileMode::read);
328  NcVar var = nc.getVar(var_name);
329  auto var_dims = var.getDims();
330  T *p_vals = nullptr;
331 
332  size_t len = 1;
333  for (const auto & dim : var_dims) {
334  len *= dim.getSize();
335  }
336 
337  try {
338  p_vals = new T[len];
339  } catch (bad_alloc & e) {
340  nc.close();
341  if (verbose_ >= 1) cout << BOLDRED <<
342  "Error: Insufficient memory when reading variable ("
343  << var_name << ")!" << RESET << endl;
344  return (INSUFFICIENT_MEMORY);
345  }
346 
347  var.getVar(p_vals);
348  vector.assign(p_vals, p_vals + len);
349 
350  nc.close();
351  delete [] p_vals;
352  return (SUCCESS);
353  };
