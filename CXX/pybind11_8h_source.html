258  void initialize_generic(unique_function_record &&unique_rec, const char *text,

260  // Do NOT receive `unique_rec` by value. If this function fails to move out the unique_ptr,
261  // we do not want this to destuct the pointer. `initialize` (the caller) still relies on the
262  // pointee being alive after this call. Only move out if a `capsule` is going to keep it alive.
263  auto rec = unique_rec.get();
264 
265  // Keep track of strdup'ed strings, and clean them up as long as the function's capsule
266  // has not taken ownership yet (when `unique_rec.release()` is called).
267  // Note: This cannot easily be fixed by a `unique_ptr` with custom deleter, because the strings
268  // are only referenced before strdup'ing. So only *after* the following block could `destruct`
269  // safely be called, but even then, `repr` could still throw in the middle of copying all strings.
270  strdup_guard guarded_strdup;
271 
272  /* Create copies of all referenced C-style strings */
273  rec->name = guarded_strdup(rec->name ? rec->name : "");
274  if (rec->doc) rec->doc = guarded_strdup(rec->doc);
275  for (auto &a: rec->args) {
276  if (a.name)
277  a.name = guarded_strdup(a.name);
278  if (a.descr)
279  a.descr = guarded_strdup(a.descr);
280  else if (a.value)
281  a.descr = guarded_strdup(repr(a.value).cast<std::string>().c_str());
282  }

285 
286 #if !defined(NDEBUG) && !defined(PYBIND11_DISABLE_NEW_STYLE_INIT_WARNING)
287  if (rec->is_constructor && !rec->is_new_style_constructor) {
288  const auto class_name = detail::get_fully_qualified_tp_name((PyTypeObject *) rec->scope.ptr());
289  const auto func_name = std::string(rec->name);

301  std::string signature;
302  size_t type_index = 0, arg_index = 0;
303  for (auto *pc = text; *pc != '\0'; ++pc) {
304  const auto c = *pc;
305 
306  if (c == '{') {
307  // Write arg name for everything except *args and **kwargs.
308  if (*(pc + 1) == '*')
309  continue;
310  // Separator for keyword-only arguments, placed before the kw
311  // arguments start
312  if (rec->nargs_kw_only > 0 && arg_index + rec->nargs_kw_only == args)
313  signature += "*, ";
314  if (arg_index < rec->args.size() && rec->args[arg_index].name) {
315  signature += rec->args[arg_index].name;

320  }

323  // Write default value if available.
324  if (arg_index < rec->args.size() && rec->args[arg_index].descr) {
325  signature += " = ";
326  signature += rec->args[arg_index].descr;
327  }

329  // argument, rather than before like *

346  rec->scope.attr("__module__").cast<std::string>() + "." +

367  rec->name = guarded_strdup("__nonzero__");
368  }

374  if (rec->sibling && PYBIND11_INSTANCE_METHOD_CHECK(rec->sibling.ptr()))

553  handle parent = n_args_in > 0 ? PyTuple_GET_ITEM(args_in, 0) : nullptr,
554  result = PYBIND11_TRY_NEXT_OVERLOAD;
555 
556  auto self_value_and_holder = value_and_holder();
557  if (overloads->is_constructor) {
558  if (!parent || !PyObject_TypeCheck(parent.ptr(), (PyTypeObject *) overloads->scope.ptr())) {
559  PyErr_SetString(PyExc_TypeError, "__init__(self, ...) called with invalid or missing `self` argument");
560  return nullptr;
561  }
562 
563  const auto tinfo = get_type_info((PyTypeObject *) overloads->scope.ptr());
564  const auto pi = reinterpret_cast<instance *>(parent.ptr());
565  self_value_and_holder = pi->get_value_and_holder(tinfo, true);
566 
567  // If this value is already registered it must mean __init__ is invoked multiple times;
568  // we really can't support that in C++, so just ignore the second __init__.
569  if (self_value_and_holder.instance_registered())
570  return none().release().ptr();
571  }
572 
573  try {
574  // We do this in two passes: in the first pass, we load arguments with `convert=false`;
575  // in the second, we allow conversion (except for arguments with an explicit
576  // py::arg().noconvert()). This lets us prefer calls without conversion, with
577  // conversion as a fallback.
578  std::vector<function_call> second_pass;
579 
580  // However, if there are no overloads, we can just skip the no-convert pass entirely
581  const bool overloaded = it != nullptr && it->next != nullptr;
582 
583  for (; it != nullptr; it = it->next) {
584 
585  /* For each overload:
586  1. Copy all positional arguments we were given, also checking to make sure that
587  named positional arguments weren't *also* specified via kwarg.
588  2. If we weren't given enough, try to make up the omitted ones by checking
589  whether they were provided by a kwarg matching the `py::arg("name")` name. If
590  so, use it (and remove it from kwargs; if not, see if the function binding
591  provided a default that we can use.
592  3. Ensure that either all keyword arguments were "consumed", or that the function
593  takes a kwargs argument to accept unconsumed kwargs.
594  4. Any positional arguments still left get put into a tuple (for args), and any
595  leftover kwargs get put into a dict.
596  5. Pack everything into a vector; if we have py::args or py::kwargs, they are an
597  extra tuple or dict at the end of the positional arguments.
598  6. Call the function call dispatcher (function_record::impl)
599 
600  If one of these fail, move on to the next overload and keep trying until we get a
601  result other than PYBIND11_TRY_NEXT_OVERLOAD.
602  */
603 
604  const function_record &func = *it;
605  size_t num_args = func.nargs; // Number of positional arguments that we need
606  if (func.has_args) --num_args; // (but don't count py::args
607  if (func.has_kwargs) --num_args; // or py::kwargs)
608  size_t pos_args = num_args - func.nargs_kw_only;
609 
610  if (!func.has_args && n_args_in > pos_args)
611  continue; // Too many positional arguments for this overload
612 
613  if (n_args_in < pos_args && func.args.size() < pos_args)
614  continue; // Not enough positional arguments given, and not enough defaults to fill in the blanks
615 
616  function_call call(func, parent);
617 
618  size_t args_to_copy = (std::min)(pos_args, n_args_in); // Protect std::min with parentheses
619  size_t args_copied = 0;
620 
621  // 0. Inject new-style `self` argument
622  if (func.is_new_style_constructor) {
623  // The `value` may have been preallocated by an old-style `__init__`
624  // if it was a preceding candidate for overload resolution.
625  if (self_value_and_holder)
626  self_value_and_holder.type->dealloc(self_value_and_holder);
627 
628  call.init_self = PyTuple_GET_ITEM(args_in, 0);
629  call.args.emplace_back(reinterpret_cast<PyObject *>(&self_value_and_holder));
630  call.args_convert.push_back(false);
631  ++args_copied;
632  }
633 
634  // 1. Copy any position arguments given.
635  bool bad_arg = false;
636  for (; args_copied < args_to_copy; ++args_copied) {
637  const argument_record *arg_rec = args_copied < func.args.size() ? &func.args[args_copied] : nullptr;
638  if (kwargs_in && arg_rec && arg_rec->name && PyDict_GetItemString(kwargs_in, arg_rec->name)) {
639  bad_arg = true;
640  break;
641  }
642 
643  handle arg(PyTuple_GET_ITEM(args_in, args_copied));
644  if (arg_rec && !arg_rec->none && arg.is_none()) {
645  bad_arg = true;
646  break;
647  }
648  call.args.push_back(arg);
649  call.args_convert.push_back(arg_rec ? arg_rec->convert : true);
650  }
651  if (bad_arg)
652  continue; // Maybe it was meant for another overload (issue #688)
653 
654  // We'll need to copy this if we steal some kwargs for defaults
655  dict kwargs = reinterpret_borrow<dict>(kwargs_in);
656 
657  // 1.5. Fill in any missing pos_only args from defaults if they exist
658  if (args_copied < func.nargs_pos_only) {
659  for (; args_copied < func.nargs_pos_only; ++args_copied) {
660  const auto &arg_rec = func.args[args_copied];
661  handle value;
662 
663  if (arg_rec.value) {
664  value = arg_rec.value;
665  }
666  if (value) {
667  call.args.push_back(value);
668  call.args_convert.push_back(arg_rec.convert);
669  } else
670  break;
671  }
672 
673  if (args_copied < func.nargs_pos_only)
674  continue; // Not enough defaults to fill the positional arguments
675  }
676 
677  // 2. Check kwargs and, failing that, defaults that may help complete the list
678  if (args_copied < num_args) {
679  bool copied_kwargs = false;
680 
681  for (; args_copied < num_args; ++args_copied) {
682  const auto &arg_rec = func.args[args_copied];
683 
684  handle value;
685  if (kwargs_in && arg_rec.name)
686  value = PyDict_GetItemString(kwargs.ptr(), arg_rec.name);
687 
688  if (value) {
689  // Consume a kwargs value
690  if (!copied_kwargs) {
691  kwargs = reinterpret_steal<dict>(PyDict_Copy(kwargs.ptr()));
692  copied_kwargs = true;
693  }
694  PyDict_DelItemString(kwargs.ptr(), arg_rec.name);
695  } else if (arg_rec.value) {
696  value = arg_rec.value;
697  }
698 
699  if (!arg_rec.none && value.is_none()) {
700  break;
701  }
702 
703  if (value) {
704  call.args.push_back(value);
705  call.args_convert.push_back(arg_rec.convert);
706  }
707  else
708  break;
709  }
710 
711  if (args_copied < num_args)
712  continue; // Not enough arguments, defaults, or kwargs to fill the positional arguments
713  }
714 
715  // 3. Check everything was consumed (unless we have a kwargs arg)
716  if (kwargs && !kwargs.empty() && !func.has_kwargs)
717  continue; // Unconsumed kwargs, but no py::kwargs argument to accept them
718 
719  // 4a. If we have a py::args argument, create a new tuple with leftovers
720  if (func.has_args) {
721  tuple extra_args;
722  if (args_to_copy == 0) {
723  // We didn't copy out any position arguments from the args_in tuple, so we
724  // can reuse it directly without copying:
725  extra_args = reinterpret_borrow<tuple>(args_in);
726  } else if (args_copied >= n_args_in) {
727  extra_args = tuple(0);
728  } else {
729  size_t args_size = n_args_in - args_copied;
730  extra_args = tuple(args_size);
731  for (size_t i = 0; i < args_size; ++i) {
732  extra_args[i] = PyTuple_GET_ITEM(args_in, args_copied + i);
733  }
734  }
735  call.args.push_back(extra_args);
736  call.args_convert.push_back(false);
737  call.args_ref = std::move(extra_args);
738  }
739 
740  // 4b. If we have a py::kwargs, pass on any remaining kwargs
741  if (func.has_kwargs) {
742  if (!kwargs.ptr())
743  kwargs = dict(); // If we didn't get one, send an empty one
744  call.args.push_back(kwargs);
745  call.args_convert.push_back(false);
746  call.kwargs_ref = std::move(kwargs);
747  }
748 
749  // 5. Put everything in a vector. Not technically step 5, we've been building it
750  // in `call.args` all along.
751  #if !defined(NDEBUG)
752  if (call.args.size() != func.nargs || call.args_convert.size() != func.nargs)
753  pybind11_fail("Internal error: function call dispatcher inserted wrong number of arguments!");
754  #endif
755 
756  std::vector<bool> second_pass_convert;
757  if (overloaded) {
758  // We're in the first no-convert pass, so swap out the conversion flags for a
759  // set of all-false flags. If the call fails, we'll swap the flags back in for
760  // the conversion-allowed call below.
761  second_pass_convert.resize(func.nargs, false);
762  call.args_convert.swap(second_pass_convert);
763  }
764 
765  // 6. Call the function.
766  try {
767  loader_life_support guard{};
768  result = func.impl(call);
769  } catch (reference_cast_error &) {
770  result = PYBIND11_TRY_NEXT_OVERLOAD;
771  }
772 
773  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD)
774  break;
775 
776  if (overloaded) {
777  // The (overloaded) call failed; if the call has at least one argument that
778  // permits conversion (i.e. it hasn't been explicitly specified `.noconvert()`)
779  // then add this call to the list of second pass overloads to try.
780  for (size_t i = func.is_method ? 1 : 0; i < pos_args; i++) {
781  if (second_pass_convert[i]) {
782  // Found one: swap the converting flags back in and store the call for
783  // the second pass.
784  call.args_convert.swap(second_pass_convert);
785  second_pass.push_back(std::move(call));
786  break;
787  }
788  }
789  }
790  }
791 
792  if (overloaded && !second_pass.empty() && result.ptr() == PYBIND11_TRY_NEXT_OVERLOAD) {
793  // The no-conversion pass finished without success, try again with conversion allowed
794  for (auto &call : second_pass) {
795  try {
796  loader_life_support guard{};
797  result = call.func.impl(call);
798  } catch (reference_cast_error &) {
799  result = PYBIND11_TRY_NEXT_OVERLOAD;
800  }
801 
802  if (result.ptr() != PYBIND11_TRY_NEXT_OVERLOAD) {
803  // The error reporting logic below expects 'it' to be valid, as it would be
804  // if we'd encountered this failure in the first-pass loop.
805  if (!result)
806  it = &call.func;
807  break;
808  }
809  }
810  }
811  } catch (error_already_set &e) {
812  e.restore();
813  return nullptr;
814 #ifdef __GLIBCXX__
815  } catch ( abi::__forced_unwind& ) {
816  throw;
817 #endif
818  } catch (...) {
819  /* When an exception is caught, give each registered exception
820  translator a chance to translate it to a Python exception
821  in reverse order of registration.
822 
823  A translator may choose to do one of the following:
824 
825  - catch the exception and call PyErr_SetString or PyErr_SetObject
826  to set a standard (or custom) Python exception, or
827  - do nothing and let the exception fall through to the next translator, or
828  - delegate translation to the next translator by throwing a new type of exception. */
829 
830  auto last_exception = std::current_exception();
831  auto &registered_exception_translators = get_internals().registered_exception_translators;
832  for (auto& translator : registered_exception_translators) {
833  try {
834  translator(last_exception);
835  } catch (...) {
836  last_exception = std::current_exception();
837  continue;
838  }
839  return nullptr;
840  }
841  PyErr_SetString(PyExc_SystemError, "Exception escaped from default exception translator!");
842  return nullptr;
843  }

848  "Did you forget to `#include <pybind11/stl.h>`? Or <pybind11/complex.h>,\n"

857  return handle(Py_NotImplemented).inc_ref().ptr();
858

865  msg += " "+ std::to_string(++ctr) + ". ";

870  std::string sig = it2->signature;

872  if (start < sig.size()) {
873  // End at the , for the next argument
874  size_t end = sig.find(", "), next = end + 2;
875  size_t ret = sig.rfind(" -> ");
876  // Or the ), if there is no comma:
877  if (end >= sig.size()) next = end = sig.find(')');
878  if (start < end && next < sig.size()) {
879  msg.append(sig, start, end - start);
880  msg += '(';
881  msg.append(sig, next, ret - next);
882  wrote_sig = true;
883  }
884  }
885  }
886  if (!wrote_sig) msg += it2->signature;
887 
888  msg += "\n";
889  }
890  msg += "\nInvoked with: ";

892  bool some_args = false;
893  for (size_t ti = overloads->is_constructor ? 1 : 0; ti < args_.size(); ++ti) {
894  if (!some_args) some_args = true;
895  else msg += ", ";

898  } catch (const error_already_set&) {
899  msg += "<repr raised Error>";
900  }
901  }
902  if (kwargs_in) {
903  auto kwargs = reinterpret_borrow<dict>(kwargs_in);
904  if (!kwargs.empty()) {
905  if (some_args) msg += "; ";
906  msg += "kwargs: ";
907  bool first = true;
908  for (auto kwarg : kwargs) {
909  if (first) first = false;
910  else msg += ", ";
911  msg += pybind11::str("{}=").format(kwarg.first);
912  try {
913  msg += pybind11::repr(kwarg.second);
914  } catch (const error_already_set&) {
915  msg += "<repr raised Error>";
916  }
917  }
918  }
919  }
920 
921  append_note_if_missing_header_is_suspected(msg);
922  PyErr_SetString(PyExc_TypeError, msg.c_str());
923  return nullptr;
924  } else if (!result) {
925  std::string msg = "Unable to convert function return value to a "
926  "Python type! The signature was\n\t";
927  msg += it->signature;
928  append_note_if_missing_header_is_suspected(msg);
929  PyErr_SetString(PyExc_TypeError, msg.c_str());
930  return nullptr;
931  } else {

933  auto *pi = reinterpret_cast<instance *>(parent.ptr());
934  self_value_and_holder.type->init_instance(pi, nullptr);

1820 inline void keep_alive_impl(handle nurse, handle patient) {
1821  if (!nurse || !patient)
1822  pybind11_fail("Could not activate keep_alive!");
1823 
1824  if (patient.is_none() || nurse.is_none())
1825  return; /* Nothing to keep alive or nothing to be kept alive by */
1826 
1827  auto tinfo = all_type_info(Py_TYPE(nurse.ptr()));
1828  if (!tinfo.empty()) {
1829  /* It's a pybind-registered type, so we can store the patient in the
1830  * internal list. */
1831  add_patient(nurse.ptr(), patient.ptr());

1835  * Boost.Python. This is not used for pybind-registered types because

1841 
1842  patient.inc_ref(); /* reference patient and leak the weak reference */
