484  size_t str_len = value.size();
485  if (str_len == 0)

488  // If we're in UTF-8 mode, we have two possible failures: one for a unicode character that
489  // is too high, and one for multiple unicode characters (caught later), so we need to figure
490  // out how long the first encoded character is in bytes to distinguish between these two
491  // errors. We also allow want to allow unicode characters U+0080 through U+00FF, as those
492  // can fit into a single char value.
493  if (StringCaster::UTF_N == 8 && str_len > 1 && str_len <= 4) {
494  auto v0 = static_cast<unsigned char>(value[0]);
495  size_t char0_bytes = !(v0 & 0x80) ? 1 : // low bits only: 0-127
496  (v0 & 0xE0) == 0xC0 ? 2 : // 0b110xxxxx - start of 2-byte sequence
497  (v0 & 0xF0) == 0xE0 ? 3 : // 0b1110xxxx - start of 3-byte sequence
498  4; // 0b11110xxx - start of 4-byte sequence
499 
500  if (char0_bytes == str_len) {
501  // If we have a 128-255 value, we can decode it into a single char:
502  if (char0_bytes == 2 && (v0 & 0xFC) == 0xC0) { // 0x110000xx 0x10xxxxxx
503  one_char = static_cast<CharT>(((v0 & 3) << 6) + (static_cast<unsigned char>(value[1]) & 0x3F));
504  return one_char;
505  }

507  throw value_error("Character code point not in range(0x100)");
508  }
509  }

512  // surrogate pair with total length 2 instantly indicates a range error (but not a "your
513  // string was too long" error).
514  else if (StringCaster::UTF_N == 16 && str_len == 2) {
515  one_char = static_cast<CharT>(value[0]);
516  if (one_char >= 0xD800 && one_char < 0xE000)
517  throw value_error("Character code point not in range(0x10000)");
518  }
519

521  throw value_error("Expected a character, but multi-character string found");
522
