540

544  auto format = spec[1].cast<tuple>()[0].cast<dtype>();

1106 inline PYBIND11_NOINLINE void register_structured_dtype(
1107  any_container<field_descriptor> fields,
1108  const std::type_info& tinfo, ssize_t itemsize,
1109  bool (*direct_converter)(PyObject *, void *&)) {
1110 
1111  auto& numpy_internals = get_numpy_internals();
1112  if (numpy_internals.get_type_info(tinfo, false))
1113  pybind11_fail("NumPy: dtype is already registered");
1114 
1115  // Use ordered fields because order matters as of NumPy 1.14:
1116  // https://docs.scipy.org/doc/numpy/release.html#multiple-field-indexing-assignment-of-structured-arrays
1117  std::vector<field_descriptor> ordered_fields(std::move(fields));
1118  std::sort(ordered_fields.begin(), ordered_fields.end(),
1119  [](const field_descriptor &a, const field_descriptor &b) { return a.offset < b.offset; });
1120 
1121  list names, formats, offsets;
1122  for (auto& field : ordered_fields) {
1123  if (!field.descr)
1124  pybind11_fail(std::string("NumPy: unsupported field dtype: `") +
1125  field.name + "` @ " + tinfo.name());
1126  names.append(PYBIND11_STR_TYPE(field.name));
1127  formats.append(field.descr);
1128  offsets.append(pybind11::int_(field.offset));
1129  }
1130  auto dtype_ptr = pybind11::dtype(names, formats, offsets, itemsize).release().ptr();
1131 
1132  // There is an existing bug in NumPy (as of v1.11): trailing bytes are
1133  // not encoded explicitly into the format string. This will supposedly
1134  // get fixed in v1.12; for further details, see these:
1135  // - https://github.com/numpy/numpy/issues/7797
1136  // - https://github.com/numpy/numpy/pull/7798
1137  // Because of this, we won't use numpy's logic to generate buffer format
1138  // strings and will just do it ourselves.
1139  ssize_t offset = 0;
1140  std::ostringstream oss;
1141  // mark the structure as unaligned with '^', because numpy and C++ don't
1142  // always agree about alignment (particularly for complex), and we're
1143  // explicitly listing all our padding. This depends on none of the fields
1144  // overriding the endianness. Putting the ^ in front of individual fields
1145  // isn't guaranteed to work due to https://github.com/numpy/numpy/issues/9049
1146  oss << "^T{";
1147  for (auto& field : ordered_fields) {
1148  if (field.offset > offset)
1149  oss << (field.offset - offset) << 'x';
1150  oss << field.format << ':' << field.name << ':';
1151  offset = field.offset + field.size;
1152  }
1153  if (itemsize > offset)
1154  oss << (itemsize - offset) << 'x';
1155  oss << '}';
1156  auto format_str = oss.str();
1157 
1158  // Sanity check: verify that NumPy properly parses our buffer format string
1159  auto& api = npy_api::get();
1160  auto arr = array(buffer_info(nullptr, itemsize, format_str, 1));
1161  if (!api.PyArray_EquivTypes_(dtype_ptr, arr.dtype().ptr()))
1162  pybind11_fail("NumPy: invalid buffer descriptor!");
1163 
1164  auto tindex = std::type_index(tinfo);
1165  numpy_internals.registered_dtypes[tindex] = { dtype_ptr, format_str };
1166  get_internals().direct_conversions[tindex].push_back(direct_converter);
1167 }

1389 enum class broadcast_trivial { non_trivial, c_trivial, f_trivial };
1390 
1391 // Populates the shape and number of dimensions for the set of buffers. Returns a broadcast_trivial
1392 // enum value indicating whether the broadcast is "trivial"--that is, has each buffer being either a
1393 // singleton or a full-size, C-contiguous (`c_trivial`) or Fortran-contiguous (`f_trivial`) storage
1394 // buffer; returns `non_trivial` otherwise.
1395 template <size_t N>
1396 broadcast_trivial broadcast(const std::array<buffer_info, N> &buffers, ssize_t &ndim, std::vector<ssize_t> &shape) {
1397  ndim = std::accumulate(buffers.begin(), buffers.end(), ssize_t(0), [](ssize_t res, const buffer_info &buf) {
1398  return std::max(res, buf.ndim);
1399  });
1400 
1401  shape.clear();
1402  shape.resize((size_t) ndim, 1);
1403 
1404  // Figure out the output size, and make sure all input arrays conform (i.e. are either size 1 or
1405  // the full size).
1406  for (size_t i = 0; i < N; ++i) {
1407  auto res_iter = shape.rbegin();
1408  auto end = buffers[i].shape.rend();
1409  for (auto shape_iter = buffers[i].shape.rbegin(); shape_iter != end; ++shape_iter, ++res_iter) {
1410  const auto &dim_size_in = *shape_iter;
1411  auto &dim_size_out = *res_iter;
1412 
1413  // Each input dimension can either be 1 or `n`, but `n` values must match across buffers
1414  if (dim_size_out == 1)
1415  dim_size_out = dim_size_in;
1416  else if (dim_size_in != 1 && dim_size_in != dim_size_out)
1417  pybind11_fail("pybind11::vectorize: incompatible size/dimension of inputs!");
1418  }
1419  }
1420 
1421  bool trivial_broadcast_c = true;
1422  bool trivial_broadcast_f = true;
1423  for (size_t i = 0; i < N && (trivial_broadcast_c || trivial_broadcast_f); ++i) {
1424  if (buffers[i].size == 1)
1425  continue;
1426 
1427  // Require the same number of dimensions:
1428  if (buffers[i].ndim != ndim)
1429  return broadcast_trivial::non_trivial;
1430 
1431  // Require all dimensions be full-size:
1432  if (!std::equal(buffers[i].shape.cbegin(), buffers[i].shape.cend(), shape.cbegin()))
1433  return broadcast_trivial::non_trivial;
1434 
1435  // Check for C contiguity (but only if previous inputs were also C contiguous)
1436  if (trivial_broadcast_c) {
1437  ssize_t expect_stride = buffers[i].itemsize;
1438  auto end = buffers[i].shape.crend();
1439  for (auto shape_iter = buffers[i].shape.crbegin(), stride_iter = buffers[i].strides.crbegin();
1440  trivial_broadcast_c && shape_iter != end; ++shape_iter, ++stride_iter) {
1441  if (expect_stride == *stride_iter)
1442  expect_stride *= *shape_iter;
1443  else
1444  trivial_broadcast_c = false;
1445  }
1446  }
1447 
1448  // Check for Fortran contiguity (if previous inputs were also F contiguous)
1449  if (trivial_broadcast_f) {
1450  ssize_t expect_stride = buffers[i].itemsize;
1451  auto end = buffers[i].shape.cend();
1452  for (auto shape_iter = buffers[i].shape.cbegin(), stride_iter = buffers[i].strides.cbegin();
1453  trivial_broadcast_f && shape_iter != end; ++shape_iter, ++stride_iter) {
1454  if (expect_stride == *stride_iter)
1455  expect_stride *= *shape_iter;
1456  else
1457  trivial_broadcast_f = false;
1458  }
1459  }
1460  }
1461 
1462  return
1463  trivial_broadcast_c ? broadcast_trivial::c_trivial :
1464  trivial_broadcast_f ? broadcast_trivial::f_trivial :
1465  broadcast_trivial::non_trivial;
1466 }
