# Reference
# https://gitlab.kitware.com/cmake/community/wikis/FAQ#can-i-do-make-uninstall-with-cmake
#
if(NOT EXISTS "${CMAKE_SOURCE_DIR}/install_manifest.txt")
    message(FATAL_ERROR "Cannot find install manifest: ${CMAKE_SOURCE_DIR}/install_manifest.txt")
endif(NOT EXISTS "${CMAKE_SOURCE_DIR}/install_manifest.txt")

file(READ "${CMAKE_SOURCE_DIR}/install_manifest.txt" files)
string(REGEX REPLACE "\n" ";" files "${files}")
foreach(file ${files})
    message(STATUS "Uninstalling ${CMAKE_INSTALL_PREFIX}/${file}")
    if(IS_SYMLINK "${CMAKE_INSTALL_PREFIX}/${file}" OR EXISTS "${CMAKE_INSTALL_PREFIX}/${file}")
        exec_program(
            "${CMAKE_COMMAND}" ARGS "-E remove \"${CMAKE_INSTALL_PREFIX}/${file}\""
            OUTPUT_VARIABLE rm_out
            RETURN_VALUE rm_retval
            )
        if(NOT "${rm_retval}" STREQUAL 0)
            message(FATAL_ERROR "Problem when removing ${CMAKE_INSTALL_PREFIX}/${file}")
        endif(NOT "${rm_retval}" STREQUAL 0)
    else(IS_SYMLINK "${CMAKE_INSTALL_PREFIX}/${file}" OR EXISTS "${CMAKE_INSTALL_PREFIX}/${file}")
        message(STATUS "File ${CMAKE_INSTALL_PREFIX}/${file} does not exist.")
    endif(IS_SYMLINK "${CMAKE_INSTALL_PREFIX}/${file}" OR EXISTS "${CMAKE_INSTALL_PREFIX}/${file}")
endforeach(file)
